nums = input("enter your array: ").strip()
arr = [int(x) for x in nums.split()]

n = len(arr)

for i in range(n):
    min_index = i
    for j in range(i+1, n):
        if arr[j] < arr[min_index]:
            min_index = j  
    arr[i], arr[min_index] = arr[min_index], arr[i]

print(f"your sorted array: {arr}")
